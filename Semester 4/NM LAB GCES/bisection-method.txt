#include <stdio.h>
#include <conio.h>
#include <math.h>
#define f(x) pow(x,3)-4*x+1
#define e 0.001

int main()
{
  float x0,x1,x2,f0,f1,f2;
  int i=0;
  printf("enter the value of x0 and x1:");
  scanf("%f %f",&x0,&x1);
  do{
     f0=f(x0);
     f1=f(x1);
     x2=(x0+x1)/2;
     f2=f(x2);
     if(f0*f2<0)
     {
         x1=x2;
     }
     else
     {
         x0=x2;
     }
     i++;
     printf("no of iteration=%d\t",i);
     printf("root=%f\t",x2);
     printf("value of function=%f\n",f2);
  }while(fabs(f2)>e);
  return 0;
    
}

{

  scanf("%f %f",&x0,&x1); //Initial value guess are stored in x1 and x2.
  do{
     f0=f(x0); //Funtional Value when x=x0
     f1=f(x1); //Functional Value when x=x1
     x2=(x0+x1)/2; //Bisectional Value which is our formula is Calculated by this Expression.
     f2=f(x2);  // functional Value when x=x2
     if(f0*f2<0)  //Condition apply 
     {
         x1=x2;
     }
     else
     {
         x0=x2;
     }
     i++;
     printf("no of iteration=%d\t",i); //Total n of iteration 
     printf("root=%f\t",x2); //Calculation of root which is x2 that we get from iteration.
     printf("value of function=%f\n",f2);
  }while(fabs(f2)>e);

}
